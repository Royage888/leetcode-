class Solution {
    public boolean isPalindrome(int x) {
        int pas = x;
        int res = 0;
        if (x < 0) {
            return false;
        } else {
            while (x != 0) {
                int t = x % 10;
                res = res * 10 + t;
                x = x / 10;
            }
            if (res == pas) {
                return true;
            } else {
               return false;
        }
    }
}
}
